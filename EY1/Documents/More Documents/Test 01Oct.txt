SELECT
  s.stud_id,
  s.surname,
  s.forename,
  s.dob,
  ah.house_no,
  ah.house_name,
  ah.apartment,
  ad.address1,
  ad.address2,
  ad.address3,
  ad.town,
  ad.post_code,
  ah.easting,
  ah.northing,
  al.address_id,
  bf.base_name,
  bf.type_id,
  ba.base_name,
  ba.type_id,
  tgi.tg_year,
  tgi.tg_desc,
  t.alloc_date

FROM 
  onelive.tg_info tgi
  inner join
  onelive.transfers t on tgi.tgh_id = t.tgh_id
  left outer join
  onelive.bases bf on t.f_base_id = bf.base_id
  left outer join
  onelive.bases ba on t.a_base_id = ba.base_id
  inner join
  onelive.addr_link al on t.stud_id = al.entity_id
  inner join
  onelive.addr_hd ah on al.address_id = ah.address_id
  inner join
  onelive.addr_det ad on ah.detail_id = ad.detail_id
  inner join
  onelive.student s on t.stud_id = s.stud_id
  



WHERE 
  tgi.tg_year = 2017

ORDER BY
  s.surname, s.forename
